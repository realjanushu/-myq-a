      int scId  = sc.nextInt();
      for(int i = 0;i<list.size();i++){

      System.out.println("请输入需要改动的水果编号id：");

      int scId = sc.nextInt();
      for(int i = 0;i<list.size();i++){

          System.out.println("请输入需要更新的水果名称：");
          l.name = sc.next();
          System.out.println("请输入需要更新的水果单价");
          l.price = sc.nextDouble();
          System.out.println("更新水果成功");
        }
      }

  }
